 private getSearchItemTemplate(data): string {
    return `
      <div>
        ${data.subtitle ? `<p>${data.subtitle}</p>` : ''}
        <p>${data.title}</p>
      </div>
    `;
  }

  private getSearchItemsTemplate(items): string {
    items.map((item) => this.getSearchItemTemplate(item)).join('');
  }


  fromEvent(this.searchCancelBtn, 'click').subscribe((e: Event) => {

      this.removeSearchModal();
      this.returnScroll();
      this.searchModalContainer.innerHTML = '';
      this.buildHistoryList(this.getHistoryList());

    });
